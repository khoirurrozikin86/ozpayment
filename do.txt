buat dto

php artisan make:dto PortfolioData --module=Portfolio \
  --fields="title:string,summary:?string,tags:?string,thumb:?file,order:?int" \
  --from-request



buat service

php artisan make:service PortfolioService --module=Portfolio --dto=PortfolioData --model=PortfolioItem


buat domain

php artisan make:domain Portfolio \
  --fields="title:string,summary:?string,tags:?string,thumb:?file,order:?int" \
  --with-model
